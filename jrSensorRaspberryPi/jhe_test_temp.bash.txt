#!/bin/bash

i=0
while [ $i -le 30 ]
do
	# Hollt die Daten vom Sensor und speichert die Zeile mit Humidity in WERTE ab
	WERTE=$(sudo ./loldht 7 | grep Humidity)
	# Temperatur und Luftfeuchtikeit in die Variablen abspeichern
	Temp=( $(echo $WERTE | awk '{ print $ 3}'))
	Hum=( $(echo $WERTE | awk '{ print $ 7}'))
	sensor='{"station":"Station1","sensor":"Sensor1","value":'${Temp}',"unit":'${Hum}'}'
	# nur zum testen
	# echo $WERTE
	# echo $Temp
	# echo $Hum
	# echo $sensor
	# Web Service mit dem POST
	curl -X POST -H "Content-Type: application/json" --data $sensor  http://192.168.79.11:8080/jrSensorWeb/ws/sensor/
	sleep 10
done
